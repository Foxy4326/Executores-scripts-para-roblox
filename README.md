            const newItem = {
                id: Date.now(),
                title,
                description,
                type,
                downloadUrl,
                date: new Date().toLocaleDateString('pt-BR'),
                lastUpdate: new Date().toLocaleDateString('pt-BR')
            };
            
            publishedItems.push(newItem);
            localStorage.setItem('publishedItems', JSON.stringify(publishedItems));
